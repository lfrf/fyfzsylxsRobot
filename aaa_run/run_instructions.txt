import json
import sys
import urllib.request
import urllib.error

API_URL = "http://127.0.0.1:19000/v1/robot/chat_turn"


def post_turn(session_id: str, turn_id: str, mode: str, text: str):
    print()
    print("=" * 70)
    print(f"SESSION={session_id} | TURN={turn_id} | MODE={mode}")
    print(f"USER: {text}")
    print("=" * 70)

    payload = {
        "session_id": session_id,
        "turn_id": turn_id,
        "mode": mode,
        "input": {
            "type": "audio_base64",
            "audio_base64": "dummy-audio-for-text-test",
            "audio_format": "wav",
            "sample_rate": 16000,
            "channels": 1,
            "text_hint": text,
        },
    }

    body = json.dumps(payload, ensure_ascii=False).encode("utf-8")
    req = urllib.request.Request(
        API_URL,
        data=body,
        headers={"Content-Type": "application/json"},
        method="POST",
    )

    try:
        with urllib.request.urlopen(req, timeout=180) as resp:
            raw = resp.read().decode("utf-8", errors="replace")
            status = resp.status
    except urllib.error.HTTPError as e:
        raw = e.read().decode("utf-8", errors="replace")
        print("HTTP_STATUS:", e.code)
        print("RAW_RESPONSE:")
        print(raw)
        return None
    except Exception as e:
        print("REQUEST_FAILED:", repr(e))
        return None

    print("HTTP_STATUS:", status)

    try:
        data = json.loads(raw)
    except Exception as e:
        print("JSON_PARSE_FAILED:", repr(e))
        print("RAW_RESPONSE:")
        print(raw)
        return None

    debug = data.get("debug") or {}

    print("success:", data.get("success"))
    print("mode:", (data.get("mode") or {}).get("mode_id"), "/", (data.get("mode") or {}).get("display_name"))
    print("mode_changed:", data.get("mode_changed"))
    print("asr_text:", data.get("asr_text"))
    print("reply_text:", data.get("reply_text"))
    print("tts.audio_url:", (data.get("tts") or {}).get("audio_url"))
    print("robot_action:", data.get("robot_action"))

    print("sources:", debug.get("sources"))
    print("mode_chain:", debug.get("mode_chain"))
    print("response_policy:", debug.get("response_policy"))
    print("rag_context_used:", debug.get("rag_context_used"))
    print("rag_matched_files:", debug.get("rag_matched_files"))

    return data


def run_all():
    print("\n===== 1. Care mode =====")
    post_turn("text-care-test", "t1", "accompany", "切换为关怀模式")
    post_turn("text-care-test", "t2", "care", "我今天有点累")
    post_turn("text-care-test", "t3", "care", "我头晕胸口痛")

    print("\n===== 2. Accompany mode =====")
    post_turn("text-accompany-test", "t1", "care", "切换为陪伴模式")
    post_turn("text-accompany-test", "t2", "accompany", "我今天有点无聊")
    post_turn("text-accompany-test", "t3", "accompany", "今天感觉有点烦")

    print("\n===== 3. Learning mode =====")
    post_turn("text-learning-test", "t1", "care", "切换为学习模式")
    post_turn("text-learning-test", "t2", "learning", "帮我制定今天的学习计划")
    post_turn("text-learning-test", "t3", "learning", "考考我嵌入式")

    print("\n===== 4. Game mode: riddle =====")
    post_turn("text-game-riddle-test", "t1", "accompany", "我们来开始游戏吧")
    post_turn("text-game-riddle-test", "t2", "game", "A")
    post_turn("text-game-riddle-test", "t3", "game", "铅笔")
    post_turn("text-game-riddle-test", "t4", "game", "不玩了")
    post_turn("text-game-riddle-test", "t5", "accompany", "我今天有点无聊")

    print("\n===== 5. Game mode: word chain =====")
    post_turn("text-game-word-test", "t1", "accompany", "开始游戏")
    post_turn("text-game-word-test", "t2", "game", "B")
    post_turn("text-game-word-test", "t3", "game", "空调")
    post_turn("text-game-word-test", "t4", "game", "切换为学习模式")
    post_turn("text-game-word-test", "t5", "learning", "考考我")


if __name__ == "__main__":
    if len(sys.argv) == 1:
        run_all()
    else:
        # 用法：
        # python3 /tmp/robot_text_test.py session turn mode "文本"
        if len(sys.argv) != 5:
            print('Usage: python3 /tmp/robot_text_test.py <session_id> <turn_id> <mode> "<text>"')
            sys.exit(1)
        post_turn(sys.argv[1], sys.argv[2], sys.argv[3], sys.argv[4])
